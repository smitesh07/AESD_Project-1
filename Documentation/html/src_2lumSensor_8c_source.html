32 void lumSensorTrigger () {
33  int ret;
34  sem_wait(sem_i2c);
35  i2cCntrl(slaveAddr);
36  lux= getLum();
37  sem_post(sem_i2c);
38  if (lux<0) {
39  latestLux->sensorConnected=false;
40  enQueueForLog(ERROR, "Luminosity Sensor Disconnected!! Will retry in a few seconds..", 0);
41  gpio_set_value(USR_LED1, 1);
42  usleep(500000);
43  gpio_set_value(USR_LED1, 0);
44  }
45  else {
46  latestLux->sensorConnected=true;
47  enQueueForLog(INFO, "Luminosity: ", lux);
48  if (lux<DARKNESS_THRESHOLD && currentState==LIGHT) {
49  enQueueForLog(WARN, "Change in luminosity level. It's now dark.", DARK);
50  currentState= DARK;
51  }
52  else if (lux>=DARKNESS_THRESHOLD && currentState==DARK) {
53  enQueueForLog(WARN, "Change in luminosity level. It's now light.", LIGHT);
54  currentState= LIGHT;
55  }
56  }
57  ret = poll(pollFds.poll_fds, 1, POLL_TIMEOUT);
58  if (ret > 0) {
59  // n = read(pollFds.f, &(pollFds.value), sizeof(pollFds.value));
60  gpio_set_value(USR_LED1, 1);
61  enQueueForLog(INFO, "Luminosity Sensor Interrupt Received",0);
62  lseek(pollFds.f, 0, SEEK_SET);
63  }
64  else {
65  gpio_set_value(USR_LED1, 0);
66  lseek(pollFds.f, 0, SEEK_SET);
67  }
68  return;
69 }
